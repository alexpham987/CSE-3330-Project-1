SET @var = 'something';
SELECT CASE WHEN c.CustID = @var THEN (c.CustID, c.Name, CONCAT('$', r.TotalAmount) AS RemainingBalance)
	    	WHERE c.CustID = @var
	    WHEN c.Name = @var THEN (c.CustID, c.Name, CONCAT('$', r.TotalAmount) AS RemainingBalance)
	    	WHERE c.Name = @var
            WHEN @var LIKE CONCAT(c.Name, '%') THEN (c.CustID, c.Name, CONCAT('$', r.TotalAmount) AS RemainingBalance)
	    	WHERE @var LIKE CONCAT(c.Name, '%')
            ELSE (c.CustID, c.Name, CONCAT('$', r.TotalAmount) AS RemainingBalance)
		ORDER BY r.TotalAmount ASC END
FROM project2.customer AS c
INNER JOIN project2.rental AS r ON c.CustID = r.CustID;
